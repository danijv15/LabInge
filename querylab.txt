use BD_B03194

CREATE TABLE Cliente (
	cedula		VARCHAR(14)		PRIMARY KEY,
	nombre		VARCHAR(15)		NOT NULL,
	apellido1	VARCHAR(15)		NOT NULL,
	apellido2	VARCHAR(15),
	correo		VARCHAR(50)		NOT NULL,
	direccion	VARCHAR(100),
);


CREATE TABLE Cuenta (
	tipo			VARCHAR(10),
	numero			INTEGER				NOT NULL,
	cedulaCliente	VARCHAR(14)			NOT NULL,

	CONSTRAINT CuentaPK PRIMARY KEY (numero, cedulaCliente ASC),
	CONSTRAINT ClienteFK FOREIGN KEY (cedulaCliente) REFERENCES Cliente(cedula), 
);


CREATE TABLE Telefono (
	numero INTEGER						NOT NULL,
	cedulaCliente	VARCHAR(14)			NOT NULL,

	CONSTRAINT TelefonoPK PRIMARY KEY (numero, cedulaCliente ASC),
	CONSTRAINT ClienFK FOREIGN KEY (cedulaCliente) REFERENCES Cliente(cedula),
);
